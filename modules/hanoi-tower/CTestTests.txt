set(prefix "${MODULE}")

add_test(
    NAME ${prefix}_go
    COMMAND ${APPLICATION}
)

    LABELS "${MODULE}")

    NAME ${prefix}_bad_arguments
    COMMAND ${APPLICATION} 6 1 2 3 4 5 6
)

    PASS_REGULAR_EXPRESSION "Invalid number of arguments"
    LABELS "${MODULE}")
